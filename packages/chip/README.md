# @react-md/chip

## Installation

```sh
$ npm install --save @react-md/chip
```

## Usage

```tsx
import React, { FC } from "react";
import { Chip } from "@react-md/chip";

const Example: FC = () => <Chip id="example-chip-1">I'm a chip!</Chip>;

export default Example;
```
